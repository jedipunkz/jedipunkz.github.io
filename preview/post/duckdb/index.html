DuckDB гҒ«е…Ҙй–ҖгҒ—гҒҰгҒҝгҒҹ

DuckDB е…Ҙй–ҖиЁҳдәӢгҖӮJSON гӮ„ CSVгҖҒParquet гҒӘгҒ©гҒ®гғ•гӮЎгӮӨгғ«гӮ’зӣҙжҺҘгӮҜгӮЁгғӘгҒ§гҒҚгӮӢгҖӮNode.js з’°еўғгҒ§гҒ®еҹәжң¬зҡ„гҒӘдҪҝз”Ёж–№жі•гҒЁгҖҒAWS WAFv2 гғӯгӮ°и§ЈжһҗгҒ®е…·дҪ“дҫӢгӮ’зҙ№д»Ӣ

гҒ“гӮ“гҒ«гҒЎгҒҜгҖӮjedipunkzрҹҡҖ гҒ§гҒҷгҖӮ

д»ҠеӣһгҒҜ DuckDB гҒ«гҒӨгҒ„гҒҰзҗҶи§ЈгҒ—гҒҹгҒ“гҒЁгӮ’гҒҫгҒЁгӮҒгӮҲгҒҶгҒЁжҖқгҒ„гҒҫгҒҷгҖӮ

DuckDB гҒҜ PostgreSQL дә’жҸӣгҒ® SQL гӮ’дҪҝгҒЈгҒҰгҖҒJSON гӮ„ CSVгҖҒParquet гҒӘгҒ©гҒ®гғ•гӮЎгӮӨгғ«гӮ’зӣҙжҺҘгӮҜгӮЁгғӘгҒ§гҒҚгӮӢгғҮгғјгӮҝгғҷгғјгӮ№гӮЁгғігӮёгғігҒ§гҒҷгҖӮгӮөгғјгғҗгғјгҒ®гӮ»гғғгғҲгӮўгғғгғ—гҒҢдёҚиҰҒгҒ§гҖҒNode.js гӮ„ Python гҒӘгҒ©гҒ®гӮўгғ—гғӘгӮұгғјгӮ·гғ§гғігҒ«зө„гҒҝиҫјгӮ“гҒ§дҪҝгҒҲгӮӢгҒҹгӮҒгҖҒгғҮгғјгӮҝеҲҶжһҗгӮ„гғӯгӮ°и§ЈжһҗгҒ®з”ЁйҖ”гҒ§жіЁзӣ®гҒ•гӮҢгҒҰгҒ„гҒҫгҒҷгҖӮ

DuckDB гҒ®зү№еҫҙ#

DuckDB гҒҜд»ҘдёӢгҒ®зү№еҫҙгӮ’жҢҒгҒЎгҒҫгҒҷгҖӮ

  • зө„гҒҝиҫјгҒҝеһӢгғҮгғјгӮҝгғҷгғјгӮ№: гӮөгғјгғҗгғјдёҚиҰҒгҒ§гҖҒгғ—гғӯгӮ»гӮ№еҶ…гҒ§еӢ•дҪң
  • й«ҳйҖҹгҒӘеҲҶжһҗеҮҰзҗҶ: еҲ—жҢҮеҗ‘гӮ№гғҲгғ¬гғјгӮёгҒЁгғҷгӮҜгғҲгғ«еҢ–е®ҹиЎҢгҒ«гӮҲгӮҠ OLAP гҒ«жңҖйҒ©еҢ–
  • еӨҡж§ҳгҒӘгғ•гӮ©гғјгғһгғғгғҲеҜҫеҝң: JSONгҖҒCSVгҖҒParquet гҒӘгҒ©гӮ’зӣҙжҺҘиӘӯгҒҝиҫјгҒҝ
  • PostgreSQL дә’жҸӣ SQL: жЁҷжә–зҡ„гҒӘ SQL гҒ§ж“ҚдҪңеҸҜиғҪ
  • и»ҪйҮҸ: SQLite гҒ®гӮҲгҒҶгҒ«еҚҳдёҖгғҗгӮӨгғҠгғӘгҒ§еӢ•дҪң

з’°еўғгӮ»гғғгғҲгӮўгғғгғ—#

д»ҠеӣһгҒҜ Node.js (TypeScript) гҒ§ DuckDB гӮ’дҪҝз”ЁгҒ—гҒҫгҒҷгҖӮ

еҹәжң¬зҡ„гҒӘгӮ»гғғгғҲгӮўгғғгғ—гӮігғјгғүгҒҜд»ҘдёӢгҒ®йҖҡгӮҠгҒ§гҒҷгҖӮ :memory: гӮ’жҢҮе®ҡгҒҷгӮӢгҒ“гҒЁгҒ§гҖҒгғЎгғўгғӘдёҠгҒ«гғҮгғјгӮҝгғҷгғјгӮ№гӮ’дҪңжҲҗгҒ—гҒҫгҒҷгҖӮж°ёз¶ҡеҢ–гҒҢеҝ…иҰҒгҒӘе ҙеҗҲгҒҜгҖҒгғ•гӮЎгӮӨгғ«гғ‘гӮ№гӮ’жҢҮе®ҡгҒҷгӮӢгҒ“гҒЁгӮӮгҒ§гҒҚгҒҫгҒҷгҖӮ

// 1. WHERE еҸҘгҒ§гғ•гӮЈгғ«гӮҝгғӘгғігӮ°

const enrichedOrders = await dbAll(
  `SELECT
    o.order_id,
    o.order_date,
    c.customer_id,
    c.name as customer_name,
    c.email,
    c.country,
    o.product,
    p.category,
    o.amount as sale_price,
    p.cost,
    o.amount - p.cost as profit
  FROM 'etl/orders.json' o
  INNER JOIN 'etl/customers.json' c ON o.customer_id = c.customer_id
  INNER JOIN 'etl/products.json' p ON o.product = p.product_name
  ORDER BY o.order_date DESC`
);
console.table(enrichedOrders);

const attackPatterns = await dbAll(
  `WITH attack_details AS (
    SELECT
      timestamp,
      terminatingRuleId,
      httpRequest.clientIp as client_ip,
      httpRequest.country as country,
      httpRequest.uri as uri,
      UNNEST(terminatingRuleMatchDetails) as detail
    FROM read_json('wafv2/waf_logs.json', format='newline_delimited')
    WHERE array_length(terminatingRuleMatchDetails) > 0
  )
  SELECT
    to_timestamp(timestamp / 1000) as datetime,
    terminatingRuleId as rule,
    detail.conditionType as attack_type,
    detail.sensitivityLevel as severity,
    detail.location as location,
    client_ip as attacker_ip,
    country,
    uri as target_endpoint
  FROM attack_details
  ORDER BY timestamp DESC`
);
console.table(attackPatterns);
