func countChannelMessages(s *discordgo.Session, channelID string) (int, error) {
    count := 0
    lastID := ""

    for {
        messages, err := s.ChannelMessages(channelID, 100, lastID, "", "")
        if err != nil {
            return 0, err
        }

        if len(messages) == 0 {
            break
        }

        count += len(messages)
        lastID = messages[len(messages)-1].ID
    }

    return count, nil
}
